        /**
         * 注销登录
         * @function  signOut

        signOut() {
            let self = this;
            this.$vux.confirm.show({
                content: '确定要退出吗',
                onConfirm() {
                    let userInfo = {
                        loginname: "",
                        avatar: "",
                        id: "",
                        token: "",
                    };
                    /**
                     * 清除本地所有信息
                     */
                    localStorage.setItem("userInfo", '');
                    self.$store.dispatch('SignOut');
                    /**
                     * 返回登录页面
                     */
                    self.$router.push("/login");
                }
            })

        }
    },

}
